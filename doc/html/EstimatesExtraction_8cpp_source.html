121 {

123  double s_ang = 0;
124  double c_ang = 0;
125 
126  for (int i = 0; i < particles.cols(); ++i)
127  {
128  out_particle.head<3>() += weights(i) * particles.col(i).head<3>();
129  out_particle.middleRows<3>(3) += weights(i) * particles.col(i).middleRows<3>(3);

136  if (versor_norm >= 0.99)
